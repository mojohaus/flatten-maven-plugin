37  
38      private final Map<String, ElementHandling> name2handlingMap;
39  
40      /**
41       * The constructor.
42       */
43      public FlattenDescriptor()
44      {
45          super();
46          this.name2handlingMap = new HashMap<String, ElementHandling>();
47      }
48  
49      /**
50       * The constructor.
51       *
52       * @param descriptor is the raw {@link Xpp3Dom} from the MOJOs configuration.
53       */
54      public FlattenDescriptor( Xpp3Dom descriptor )
55      {
56          this();
57          for ( PomProperty<?> property : PomProperty.getPomProperties() )
58          {
59              String element = property.getName();
60              if ( descriptor.getChild( element ) != null )
61              {
62                  this.name2handlingMap.put( element, ElementHandling.expand );
63              }
64          }
65      }
66  
67      /**
68       * Generic method to get a {@link ElementHandling}.
69       *
70       * @param property is the {@link PomProperty} such as {@link PomProperty#NAME}.
71       * @return the {@link ElementHandling}. Will be {@link ElementHandling#flatten flattened} as fallback if undefined.
72       */
73      public ElementHandling getHandling( PomProperty<?> property )
74      {
75          ElementHandling handling = this.name2handlingMap.get( property.getName() );
76          if ( handling == null )
77          {
78              handling = ElementHandling.flatten;
79          }
80          return handling;
81      }
82  
83      /**
84       * Generic method to set an {@link ElementHandling}.
85       *
86       * @param property is the {@link PomProperty} such as {@link PomProperty#NAME}.
87       * @param handling the new {@link ElementHandling}.
88       */
89      public void setHandling( PomProperty<?> property, ElementHandling handling )
90      {
91  
92          this.name2handlingMap.put( property.getName(), handling );
93      }

495     /**
496      * Creates and returns a new {@link FlattenDescriptor} with the {@link ElementHandling}s merged from this and the
497      * given {@link FlattenDescriptor}.
498      *
499      * @param descriptor is the {@link FlattenDescriptor} to merge with this one.
500      * @return the merged {@link FlattenDescriptor}.
501      */
502     public FlattenDescriptor merge( FlattenDescriptor descriptor )
503     {
504         FlattenDescriptor result = new FlattenDescriptor();
505         for ( PomProperty<?> property : PomProperty.getPomProperties() )
506         {
507             String name = property.getName();
508             ElementHandling handling = this.name2handlingMap.get( name );
509             if ( handling == null )
510             {
511                 handling = descriptor.name2handlingMap.get( name );
512             }
513             if ( handling != null )
514             {
515                 result.name2handlingMap.put( name, handling );
516             }
517         }
518         return result;
519     }
520 
521     /**
522      * @return <code>true</code> if none of the properties has been set explicitly, <code>false</code> otherwise.
523      */
524     public boolean isEmpty()
525     {
526         for ( ElementHandling handling : this.name2handlingMap.values() )
527         {
528             if ( handling != null )
529             {
530                 return false;
531             }
532         }
533         return true;
534     }
